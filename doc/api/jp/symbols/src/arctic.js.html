 70 		    maxY = Math.max(pos0[1], pos1[1], pos2[1], pos3[1]);

196 			function overrideMethod(childMethod, parentMethod){
197 				var method = function(){
198 					var childScope = this;
199 					var $super = (parentMethod !== undefined) ? function(){ return parentMethod.apply(childScope, arguments); } : undefined;
200 					var args = copyArray(arguments);
201 					args.unshift($super);
202 					return childMethod.apply(childScope, args);
203 				}
204 				method.valueOf = bind(childMethod.valueOf, childMethod);
205 				method.toString = bind(childMethod.toString, childMethod);
206 				return method;		
207 			}
208 			
209 			if(typeof arguments[0] == 'function'){
210 				newclass.prototype = superclass;
211 				klass.prototype = new newclass;
212 				Array.prototype.shift.apply(arguments);
213 			}
214 			var def = arguments[0];
215 			for(var prop in def){
216 				var value = def[prop];
217 				if(typeof value == 'function'){
218 					if(argumentNames(value)[0] == '$super'){
219 						value = overrideMethod(value, superclass[prop]);
220 					}else if(prop == 'initialize'){
221 						if(superclass.initialize){
222 							var childInitialize = value;
223 							value = function(){
224 								superclass.initialize.apply(this, arguments);
225 								childInitialize.apply(this, arguments);
226 							}
227 						}
228 					}
229 				}
230 				klass.prototype[prop] = value;
231 			}
232 			klass.prototype.constructor = klass;
233 			
234 			return klass;
235 		}

238 			create:create
239 		};

343 		 * @param {String} type イベントのタイプ
344 		 * @param {Function} callback コールバック関数
345 		 */
346 		addEventListener:function(type, callback){
347 			var arr = EventDispatcher.listenHash[type];
348 			if(!arr){
349 				arr = EventDispatcher.listenHash[type] = [];
350 			}

356 		 * @param {String} type イベントのタイプ
357 		 * @param {Function} callback コールバック関数
358 		 */
359 		removeEventListener:function(type, callback){
360 			var arr = EventDispatcher.listenHash[type];
361 			if(!arr) return;
362 	
363 			var succeed = false;
364 	
365 			for(var i = 0; i < arr.length; i++){
366 				var obj = arr[i];
367 				if(obj.target == this){
368 					if(callback && obj.callback == callback || !callback){
369 						arr.splice(i, 1);
370 						succeed = true;
371 					}
372 				}
373 			}
374 			if(arr.length == 0){
375 				delete EventDispatcher.listenHash[type];
376 				succeed = true;
377 			}

388 			}else{
389 				e = arguments[0];

391 			}

395 			var arr = [];
396 			var tmp = EventDispatcher.listenHash[type];
397 			for(var i = 0; i < tmp.length; i++){
